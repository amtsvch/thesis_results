/**
 * Pops the top item off of this stack and return it.
 *
 * @return the top item on the stack
 * @throws EmptyStackException  if the stack is empty
 */
public E pop() throws EmptyStackException {
    int n = size();
    if (n <= 0) {
        throw new EmptyStackException();
    } else {
        return remove(n - 1);
    }
}
